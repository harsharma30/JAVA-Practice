types of loop:
-> entry control loop:
	for loop, while loop, 
-> exit control loop:
	do.. while loop,

for loop:
-> for(initialization; condition; loop counter updation);
-> for(initialization; condition; loop counter){}
-> int i= 10;
	for(; condition; loop counter){}
-> int i= 10;
	for(; ; loop counter){
	condition;
	}
-> int i= 10;
	for(; ; ){
	condition;
	i++;
	}
